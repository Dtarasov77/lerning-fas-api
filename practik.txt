Создать класс User с атрибутами name, email, age.
Методы:
  - greet() — возвращает строку приветствия
  - is_adult() — возвращает True если возраст >= 18

# Правильное выполнение:
class User:
    def __init__(self, name, age, email):
        self.name = name
        self.age = age
        self.email = email

    def greet(self):
        return f"Привет! Меня зовут {self.name}, мне {self.age} лет. Связаться: {self.email}"

    def is_adult(self):
        return self.age >= 18

###### Задание 2 — Инкапсуляция (Этап 1.2) ######
Создать класс BankAccount с приватными атрибутами:
  - __balance (начальное значение передаётся в конструктор)
  - __owner (имя владельца)
  - __history (пустой список операций)
Методы:
  - deposit(amount) — пополнить счёт, проверить что сумма > 0, записать в историю
  - withdraw(amount) — снять деньги, проверить сумма > 0 И хватает средств, записать в историю
  - balance — @property, только чтение
  - owner — @property, только чтение
  - get_history() — возвращает копию истории

# Правильное выполнение:
class BankAccount:
    def __init__(self, owner, balance):
        self.__balance = balance
        self.__owner = owner
        self.__history = []

    def deposit(self, amount):
        if amount <= 0:
            raise ValueError("Сумма должна быть больше 0")
        self.__balance += amount
        self.__history.append(f"Пополнение: +{amount} руб.")

    def withdraw(self, amount):
        if amount <= 0:
            raise ValueError("Сумма должна быть больше 0")
        if self.__balance < amount:
            raise ValueError(f"Недостаточно средств. Баланс: {self.__balance} руб.")
        self.__balance -= amount
        self.__history.append(f"Снятие: -{amount} руб.")

    @property
    def balance(self):
        return self.__balance

    @property
    def owner(self):
        return self.__owner

    def get_history(self):
        return self.__history.copy()

account = BankAccount("Дима", 1000)
account.deposit(500)
account.withdraw(200)
print(account.balance)        # 1300
print(account.owner)          # Дима
print(account.get_history())  # ['Пополнение: +500 руб.', 'Снятие: -200 руб.']
account.withdraw(5000)        # ValueError: Недостаточно средств
account.deposit(-100)         # ValueError: Сумма должна быть больше 0

# Замечания при проверке:
# 1. Использовать raise ValueError вместо молчаливого if (чтобы ошибка не проходила тихо)
# 2. Проверка amount <= 0 (не < 0), иначе deposit(0) добавит 0 руб. без ошибки
# 3. Без пробела перед скобками: def balance(self), не def balance (self)


###### Задание 3 — Наследование (Этап 1.3) ######
# Система пользователей: BaseUser → AdminUser, ModeratorUser
# Каждый тип пользователя имеет свои права (permissions).

# ===== ПОДРОБНОЕ РЕШЕНИЕ С КОММЕНТАРИЯМИ =====

class BaseUser:
    """
    Базовый класс пользователя.
    Содержит всё общее: имя, email, роль, приветствие.
    Дочерние классы будут НАСЛЕДОВАТЬ это и добавлять/менять своё.
    """

    def __init__(self, name, email):
        self.name = name
        self.email = email
        self._role = "user"
        # _role — защищённый атрибут (одно подчёркивание)
        # Дочерние классы смогут его менять, но снаружи трогать не принято

    @property
    def role(self):
        """Только чтение роли. Нет setter'а — нельзя менять снаружи."""
        return self._role

    def greet(self):
        """Базовое приветствие. Дочерние классы могут переопределить."""
        return f"Я {self.name}, роль: {self._role}"

    def has_permission(self, action):
        """
        Базовый пользователь НЕ может ничего.
        Дочерние классы переопределят этот метод — это ПОЛИМОРФИЗМ.
        action — строка, например "ban_user", "delete_post", "mute_user"
        """
        return False


class AdminUser(BaseUser):
    """
    Администратор. Наследует BaseUser.
    Добавляет: access_level (уровень доступа).
    Переопределяет: greet() и has_permission().
    """

    def __init__(self, name, email, access_level):
        # super().__init__() — вызываем __init__ РОДИТЕЛЯ (BaseUser)
        # Это установит self.name, self.email, self._role
        super().__init__(name, email)

        # Теперь МЕНЯЕМ роль на "admin" (можем, т.к. _role — защищённый)
        self._role = "admin"

        # Добавляем СВОЙ атрибут, которого нет у BaseUser
        self.access_level = access_level

    def greet(self):
        """
        ПЕРЕОПРЕДЕЛЯЕМ метод родителя (override).
        У админа приветствие отличается — показываем уровень доступа.
        """
        return f"Я админ {self.name}, уровень доступа: {self.access_level}"

    def has_permission(self, action):
        """
        ПЕРЕОПРЕДЕЛЯЕМ метод родителя.
        Логика прав зависит от уровня доступа:
          - level >= 10 → может ВСЁ
          - level >= 5  → может "delete_post" и "ban_user"
          - level < 5   → может только "delete_post"
        """
        if self.access_level >= 10:
            return True  # бог-режим, может всё
        elif self.access_level >= 5:
            return action in ("delete_post", "ban_user")
            # action in (...) возвращает True/False
            # Это короче, чем писать if action == "delete_post" or action == "ban_user"
        else:
            return action == "delete_post"


class ModeratorUser(BaseUser):
    """
    Модератор. Наследует BaseUser.
    НЕ добавляет новых атрибутов (нет своего __init__).
    Переопределяет только has_permission().
    """

    def __init__(self, name, email):
        super().__init__(name, email)
        self._role = "moderator"

    # greet() НЕ переопределяем!
    # Поэтому вызовется greet() из BaseUser → "Я Маша, роль: moderator"

    def has_permission(self, action):
        """Модератор может только мьютить и удалять посты."""
        return action in ("mute_user", "delete_post")


# ===== ПРОВЕРКА =====

admin = AdminUser("Дима", "admin@mail.ru", 10)
moder = ModeratorUser("Маша", "moder@mail.ru")
user = BaseUser("Аня", "anya@mail.ru")

print("--- Приветствия ---")
print(admin.greet())     # Я админ Дима, уровень доступа: 10
print(moder.greet())     # Я Маша, роль: moderator  (метод от BaseUser!)
print(user.greet())      # Я Аня, роль: user

print("\n--- Права доступа ---")
print(admin.has_permission("ban_user"))     # True  (level=10, может всё)
print(admin.has_permission("launch_nuke"))  # True  (level=10, может ВООБЩЕ всё)
print(moder.has_permission("ban_user"))     # False (модератор не может банить)
print(moder.has_permission("mute_user"))    # True  (модератор может мьютить)
print(moder.has_permission("delete_post"))  # True  (модератор может удалять посты)
print(user.has_permission("anything"))      # False (базовый юзер ничего не может)

print("\n--- isinstance (проверка типов) ---")
print(isinstance(admin, AdminUser))   # True — admin это AdminUser
print(isinstance(admin, BaseUser))    # True — admin ТАКЖЕ является BaseUser (наследование!)
print(isinstance(user, AdminUser))    # False — обычный user НЕ является AdminUser

print("\n--- Роли (property) ---")
print(admin.role)    # admin
print(moder.role)    # moderator
print(user.role)     # user
# admin.role = "hacker"  ← ❌ ошибка! Нет setter'а — нельзя менять

# ===== КЛЮЧЕВЫЕ МОМЕНТЫ ДЛЯ ЗАПОМИНАНИЯ =====
#
# 1. super().__init__() — ВСЕГДА вызывай в дочернем __init__,
#    иначе атрибуты родителя (name, email) не создадутся
#
# 2. Переопределение метода — просто пишешь метод с тем же именем
#    в дочернем классе. Python вызовет ДОЧЕРНЮЮ версию
#
# 3. Если метод НЕ переопределён — вызывается родительская версия
#    (как greet() у ModeratorUser)
#
# 4. isinstance(admin, BaseUser) == True — дочерний объект
#    ЯВЛЯЕТСЯ экземпляром родительского класса. Это важно!
#
# 5. action in ("a", "b", "c") — питонический способ проверить
#    "входит ли значение в набор допустимых"
